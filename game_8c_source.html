206 void gameLoop(unsigned char matrix[][SIZE], SDL_Renderer *gRenderer)
207 {
208  TTF_Font *font = NULL;
209  font = TTF_OpenFont(FONT_PATH, CELL_FONT_SIZE);
210  if (font == NULL)
211  {
212  perror("The required font was not found");

214  }
215 
216  render_game(gRenderer, matrix, font);
217 
218  BOOL quit = 0;
219  SDL_Event e;
220  while (!quit)
221  {
222  while (SDL_PollEvent(&e) != 0)
223  {
224  if (e.type == SDL_QUIT)
225  {
226  quit = 1;
227  }
228  else if (e.type == SDL_KEYUP)
229  {
230  handle_move(e, matrix, gRenderer);
231 
232  render_game(gRenderer, matrix, font);
233  }
234  else if (e.type == SDL_MOUSEBUTTONUP)
235  {
236  button_action(e, matrix);
237  render_game(gRenderer, matrix, font);
238  }
239  }
240  }
241  TTF_CloseFont(font);
242  //无需清空字体。
